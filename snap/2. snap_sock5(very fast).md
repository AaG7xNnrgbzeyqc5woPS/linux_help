# 1. snap_proxy sock5
  set sock5 proxy:  
   
    sudo snap set system proxy.http=socks5://127.0.0.1:1080
    sudo snap set system proxy.https=socks5://127.0.0.1:1080
    
  check:  
   
    sudo snap get system
    sudo snap get system proxy
